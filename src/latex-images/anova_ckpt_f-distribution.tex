\begin{tikzpicture}[
    declare function={
        gamma(\z)=2.506628274631*sqrt(1/\z)+ 0.20888568*(1/\z)^(1.5)+ 0.00870357*(1/\z)^(2.5)- (174.2106599*(1/\z)^(3.5))/25920- (715.6423511*(1/\z)^(4.5))/1244160)*exp((-ln(1/\z)-1)*\z;
    },
    declare function={
        beta(\x,\y)=gamma(\x)*gamma(\y)/gamma(\x+\y);
    },
    declare function={
        fdist(\x,\a,\b) = 1 / beta(\a/2, \b/2) * (\a/\b)^(\a/2) * \x^(\a/2-1) * (1 + \a/\b*\x)^(-(\a + \b)/2);
    }
  ]
    \begin{axis}[
        axis lines*=middle,
        no markers,
        domain=0.01:4,
        samples=500,
        xmin = 0, xmax = 4,
        ymin = -0.05, ymax = 1,
        xtick={0,1,2,3},
        ticklabel style={font=\scriptsize},
        ytick=\empty,
        grid = none,
        axis on top,
        height=5cm,
        width=8cm
      ]
      \addplot+[thick,blue] { fdist(x, 10,10) };
    \end{axis}
\end{tikzpicture}
